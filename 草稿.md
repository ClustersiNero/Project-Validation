from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
from reportlab.lib.styles import getSampleStyleSheet

doc = SimpleDocTemplate("/mnt/data/BlockC_5Day_Plan.md")

styles = getSampleStyleSheet()
content = []

def add(title, items):
    content.append(Paragraph(f"<b>{title}</b>", styles['Normal']))
    content.append(Spacer(1, 8))
    for i in items:
        content.append(Paragraph(f"- {i}", styles['Normal']))
    content.append(Spacer(1, 16))

add("Day 1 — System Overview", [
    "validation-first system",
    "pipeline structure",
    "layer separation",
    "determinism",
    "auditability"
])

add("Day 2 — Engine & RNG", [
    "deterministic RNG",
    "seed control",
    "no runtime manipulation",
    "predefined outcome space",
    "sampling process"
])

add("Day 3 — Canonical Layer", [
    "single source of truth",
    "replayability",
    "state recording",
    "no interpretation",
    "data completeness"
])

add("Day 4 — Metrics Layer", [
    "descriptive metrics",
    "RTP",
    "hit frequency",
    "distribution",
    "no judgment"
])

add("Day 5 — Validation Layer", [
    "structural validation",
    "statistical validation",
    "confidence intervals",
    "regression checks",
    "explicit verdicts"
])

doc.build(content)
